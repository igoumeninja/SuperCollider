/* 

Simple way for playing and displaying a Fibonacci tree. For rehearsing the Stasis performance. 

*/


// Calculating the duration from the beats: 


l = Fib.ascending(15).flat.size; // The length of a Fibonacci tree of depth 15 is 1597

(l / 1.30) / 60; // The duration is 20.474358974359 minutes


// Ascending tree: Short sections first

(

d = FibDisplay.new;

d.show;


t = { c = TempoClock(1.3); };

a = SyncSender(clockFunc: t);

a.pattern = Pfib(Fib.ascending(15)).asPbind(a);

a.start;

)


// Descending tree: Long sections first

(

d = FibDisplay(oscMessage: 'l_branch');

d.show;
